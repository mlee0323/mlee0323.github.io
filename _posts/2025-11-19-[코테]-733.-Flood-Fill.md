class Solution:
    def floodFill(self, image: List[List[int]], sr: int, sc: int, color: int) -> List[List[int]]:
        
        # 1. 지도의 크기를 잽니다 (나중에 지도 밖으로 나가는지 체크하려고)
        R, C = len(image), len(image[0])
        
        # 2. 원래 무슨 색이었는지 기억해둡니다 (예: 1)
        old_color = image[sr][sc]
        
        # 만약 바꾸려는 색이 원래 색이랑 같으면 아무것도 안 해도 됩니다.
        if old_color == color:
            return image

        # --- 여기가 로봇(함수) 정의입니다 ---
        def dfs(r, c):
            # [규칙 1] 현재 위치가 지도 밖이거나, 색깔이 old_color가 아니면 멈춤!
            # (다른 색깔이거나 이미 색칠한 곳이면 안 갑니다)
            if image[r][c] != old_color:
                return
            
            # [규칙 2] 색칠하기!
            image[r][c] = color
            
            # [규칙 3] 상하좌우 살펴보고 출동하기 (재귀 호출)
            # 위로 출동 (r-1)
            if r - 1 >= 0: dfs(r - 1, c)
            # 아래로 출동 (r+1)
            if r + 1 < R:  dfs(r + 1, c)
            # 왼쪽으로 출동 (c-1)
            if c - 1 >= 0: dfs(r, c - 1)
            # 오른쪽으로 출동 (c+1)
            if c + 1 < C:  dfs(r, c + 1)
        
        # --- 로봇 가동 시작! ---
        dfs(sr, sc)
        
        return image
