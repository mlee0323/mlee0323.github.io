sudo apt install openssh-server -y

- ssh 서버 설치

usermod -d &lt;home dir&gt; <username>


sudo useradd msp05-1 

- 새로운 user 추가
- sudo passwd msp05-1
	- 비밀번호 업데이트
- sudo mkdir /home/msp05-1
	- home 에 디렉토리 생성
- sudo chown -R msp05-1:msp05-1 /home/msp05-1
	- user owner 변경
- sudo chmod 750 /home/msp05-1
	- 권한부여

cmd에서 ssh 접속

- ssh [msp05-1@192.168.184.129](mailto:msp05-1@192.168.184.129)

user 삭제

- sudo userdel msp05-1
	- 이건 home 에 있는 파일 삭제 x
- sudo userdel -r msp05-2
- sudo deluser --remove-home msp05-3

usermod -l &lt;변경될 이름&gt; &lt;기존 이름&gt;


sudo groupadd &lt;이름&gt;


sudo usermode -g &lt;그룹이름&gt; &lt;추가 사용자 이름&gt;

- 단, 사용자 그룹이 바뀜

sudo usermode -G &lt;그룹이름&gt; &lt;추가 사용자 이름&gt;

- 그룹을 추가해줌

sudo usermode -e &lt;날짜&gt; &lt;계정&gt;

- 날짜로 계정 만료일 지정

 sudo usermod -L &lt;계정&gt;

- 계정 잠금

 sudo usermod -U &lt;계정&gt;

- 잠금 해제
