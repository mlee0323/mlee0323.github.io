```python
len(a) # 객체의 수, 또는 길이

# range문법. 만약 안에 두개가 있으면 a, b 까지. 하나만 있으면 0 에서 a까지
range(a, b)
range(a)

mapping = {")":"(", "}":"{", "]":"["} # dictionary 문법으로 key:value

mapping.values() # value 값 선택
mapping.keys() # key 값 선택

stack.append(a) # 스택에 a를 쌓는것
stack.pop() # 스택에서 pop

# min, max 안에 위치는 상관없이 둘이 비교한다
min(a, b)
max(a, b)

str.isalnum() # str이 알파벳/숫자인지 확인
str.lower() # 알파벳을 소문자로 변환

sorted(a) # a의 문자열을 정렬
```
