class Solution:
    def search(self, nums: List[int], target: int) -> int:
        numMap = {}
        n = len(nums)

        for i in range(n):
            numMap[nums[i]] = i
        
        for i in range(n):
            if target in numMap:
                return numMap[target]
            else:
                return (-1)

#### Binary Search Solution



{% raw %}
```python
class Solution:
    def search(self, nums: List[int], target: int) -> int:
        left = 0
        right = len(nums) - 1

        while left <= right:
            mid = (left + right) // 2

            if nums[mid] == target:
                return mid
            elif nums[mid] < target:
                left = mid + 1
            else:
                right = mid - 1
            
        return -1
```
